-- KI·ªÇM TRA QUY·ªÄN
function IDSystem:CheckPermission()
    local userID = self.UserID

    -- Ki·ªÉm tra OWNER
    if userID == self.OwnerID then
        self.IsOwner = true
        self.IsAdmin = true
        self.IsVIP = true
        self.IsUser = false
        return "OWNER"
    end

    -- Ki·ªÉm tra ADMIN
    if self.AdminIDs[userID] then
        self.IsAdmin = true
        self.IsVIP = true
        self.IsUser = false
        return "ADMIN"
    end

    -- Ki·ªÉm tra VIP
    if self.VIPIDs[userID] then
        self.IsVIP = true
        self.IsUser = false
        return "VIP"
    end

    -- USER m·∫∑c ƒë·ªãnh
    self.IsUser = true
    return "USER"
end

    requiredRank = requiredRank or "USER"

    -- OWNER ƒë∆∞·ª£c l√†m m·ªçi th·ª©

    -- ADMIN ƒë∆∞·ª£c l√†m m·ªçi th·ª© tr·ª´ OWNER

    end

    if self.IsVIP then
        if requiredRank == "OWNER" or requiredRank == "ADMIN" then

    if self.IsUser then
