-- KI·ªÇM TRA QUY·ªÄN
function IDSystem:CheckPermission()
    local userID = self.UserID

    if userID == self.OwnerID then
        self.IsOwner = true; self.IsAdmin = true; self.IsVIP = true; self.IsUser = false
        return "OWNER"
    end
    if self.AdminIDs[userID] then
        self.IsAdmin = true; self.IsVIP = true; self.IsUser = false
        return "ADMIN"
    end
    if self.VIPIDs[userID] then
        self.IsVIP = true; self.IsUser = false
        return "VIP"
    end
    self.IsUser = true
    return "USER"
end

-- KI·ªÇM TRA QUY·ªÄN TRUY C·∫¨P T√çNH NƒÇNG
function IDSystem:CanAccess(featureID, requiredRank)
    requiredRank = requiredRank or "USER"

    if self.IsOwner then return true end
    if self.IsAdmin then return requiredRank ~= "OWNER" end
    if self.IsVIP then return requiredRank ~= "OWNER" and requiredRank ~= "ADMIN" end
    if self.IsUser then return requiredRank == "USER" end

    return false
end
