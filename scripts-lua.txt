function IDSystem:CheckPermission()
    local userID = self.UserID
    if userID == self.OwnerID then
        self.Rank = "OWNER"
    elseif self.AdminIDs[userID] then
        self.Rank = "ADMIN"
    elseif self.VIPIDs[userID] then
        self.Rank = "VIP"
    else
        self.Rank = "USER"
    end
    return self.Rank
end

IDSystem:CheckPermission()

function IDSystem:CanAccess(requiredRank)
    requiredRank = requiredRank or "USER"
    return RankLevel[self.Rank] >= RankLevel[requiredRank]
end
