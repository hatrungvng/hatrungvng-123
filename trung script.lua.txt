local function GetNearestPlayer()
    local nearest = nil
    local shortest = 9999
    local character = player.Character
    local root = character and character:FindFirstChild("HumanoidRootPart")
    
    if not root then return nil end
    
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= player and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            local vRoot = v.Character.HumanoidRootPart
            local vHum = v.Character:FindFirstChild("Humanoid")
            
            if vHum and vHum.Health > 0 then
                local dist = (root.Position - vRoot.Position).Magnitude
                if dist < shortest then
                    nearest = v
                    shortest = dist
                end
            end
        end
    end
    
    return nearest
end

---------------------------------------------------------
-- HÃ€M Láº¤Y NGÆ¯á»œI CHÆ I MÃU THáº¤P NHáº¤T
---------------------------------------------------------
local function GetLowestHpPlayer()
    local target = nil
    local lowestHp = math.huge
    local character = player.Character
    local root = character and character:FindFirstChild("HumanoidRootPart")
    
    if not root then return nil end
    
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= player and v.Character and v.Character:FindFirstChild("Humanoid") then
            local vHum = v.Character.Humanoid
            local vRoot = v.Character:FindFirstChild("HumanoidRootPart")
            
            if vHum and vHum.Health > 0 and vRoot then
                local dist = (root.Position - vRoot.Position).Magnitude
                if dist < 300 and vHum.Health < lowestHp then
                    target = v
                    lowestHp = vHum.Health
                end
            end
        end
    end
    
    return target
end

    -- TEXT KHOáº¢NG CÃCH

        -- Váº½ mÅ©i tÃªn á»Ÿ rÃ¬a mÃ n hÃ¬nh

            math.clamp(arrowPos.X, 60, screenSize.X - 60),

            GhimTam.TargetArrow.Visible = false
        end
    end

            camera.CFrame = camera.CFrame:Lerp(
                CFrame.new(camera.CFrame.p, targetPart.Position),
                smoothness
            )
        end
    end
end)
